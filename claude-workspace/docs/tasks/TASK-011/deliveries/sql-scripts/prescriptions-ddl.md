CREATE TABLE prescription (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    clinic_id       UUID NOT NULL REFERENCES clinic(id) ON DELETE RESTRICT,
    visit_id        UUID,                                     -- nullable; visit FK enforced at app level
    doctor_id       UUID NOT NULL,                            -- references user.id at app level
    dispense_type   TEXT NOT NULL DEFAULT 'in_house'
                    CHECK (dispense_type IN ('in_house','external','mixed')),
    status          TEXT NOT NULL DEFAULT 'draft'
                    CHECK (status IN ('draft','pending','dispensed','cancelled')),
    notes           TEXT,
    prescribed_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
    cancelled_at    TIMESTAMPTZ,
    cancel_reason   TEXT,
    -- BaseEntity
    is_deleted      BOOLEAN NOT NULL DEFAULT false,
    deleted_at      TIMESTAMPTZ,
    deleted_by      UUID,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    created_by      UUID,
    updated_by      UUID,
    version         INTEGER NOT NULL DEFAULT 1
);

-- Indexes
CREATE INDEX ix_prescription_clinic_id ON prescription(clinic_id);
CREATE INDEX ix_prescription_clinic_visit ON prescription(clinic_id, visit_id);
CREATE INDEX ix_prescription_clinic_doctor_prescribed
    ON prescription(clinic_id, doctor_id, prescribed_at DESC)
    WHERE is_deleted = false;

-- RLS
ALTER TABLE prescription ENABLE ROW LEVEL SECURITY;
CREATE POLICY rls_prescription_clinic_isolation ON prescription
    USING (clinic_id::text = current_setting('app.current_clinic_id', true));
GRANT SELECT, INSERT, UPDATE, DELETE ON prescription TO cms_app;

CREATE TABLE prescription_item (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    clinic_id           UUID NOT NULL REFERENCES clinic(id) ON DELETE RESTRICT,
    prescription_id     UUID NOT NULL REFERENCES prescription(id) ON DELETE CASCADE,
    medicine_id         UUID REFERENCES medicine(id) ON DELETE RESTRICT,  -- NULL for external
    medicine_name       TEXT NOT NULL,          -- snapshot from catalog, or free-text
    dosage              TEXT NOT NULL,          -- e.g. "1 viên × 3 lần / ngày"
    quantity            NUMERIC NOT NULL        CHECK (quantity > 0),
    unit                VARCHAR(50) NOT NULL,
    dispense_source     TEXT NOT NULL           CHECK (dispense_source IN ('in_house','external')),
    in_house_status     TEXT                    CHECK (in_house_status IN ('reserved','dispensed','released') OR in_house_status IS NULL),
    unit_price          NUMERIC(15,2),
    sort_order          INTEGER DEFAULT 0,
    -- BaseEntity
    is_deleted          BOOLEAN NOT NULL DEFAULT false,
    deleted_at          TIMESTAMPTZ,
    deleted_by          UUID,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    created_by          UUID,
    updated_by          UUID,
    version             INTEGER NOT NULL DEFAULT 1
);

-- Indexes
CREATE INDEX ix_prescription_item_clinic_id ON prescription_item(clinic_id);
CREATE INDEX ix_prescription_item_clinic_prescription ON prescription_item(clinic_id, prescription_id);
CREATE INDEX ix_prescription_item_medicine_id ON prescription_item(medicine_id);

-- RLS

GRANT SELECT, INSERT, UPDATE, DELETE ON prescription_item TO cms_app;
